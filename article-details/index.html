# 8.4 请求文章的详情数据

  1. 在 /src/api/ 目录下新建 articleAPI.js 模块：
import request from '@/utils/request.js'

// 获取文章详情的 API（形参中的 id 是文章的 id）
export const getArticleDetailAPI = id => {
  return request.get(`/v1_0/articles/${id}`)
}

data() {

  1. 在 ArticleDetail.vue 组件的 methods 中定义名为 initArticle 的方法：
methods: {

    // 1. 调用 API 接口
    const { data: res } = await getArticleDetailAPI(this.id)

# 8.6 实现关注和取消关注的功能

# 8.6.1 实现关注功能

  1. 在 /src/api/articleAPI.js 模块中，定义名为 followUserAPI 的接口：
// 关注用户的 API（形参中的 userId 是文字作者的 id）
export const followUserAPI = userId => {
  return request.post('/v1_0/user/followings', {
    target: userId
  })
}
1
2
3
4
5
6
  1. 在 ArticleDetail.vue 组件中按需导入 followUserAPI 方法：
// 按需导入 API 接口
import { getArticleDetailAPI, followUserAPI } from '@/api/articleAPI.js'
1
2
  1. 在 ArticleDetail.vue 组件的 methods 节点下声明 setFollow 的方法:
methods: {
  // 关注作者
  async setFollow() {
    // 调用关注作者的 API 接口
    const { data: res } = await followUserAPI(this.article.aut_id.toString())
    if (res.message === 'OK') {
      // 提示用户成功
      this.$toast.success('关注成功！')
      // 手动更改关注的状态
      this.article.is_followed = true
    }
  }
}

# 8.6.2 实现取消关注功能

  1. 在 /src/api/articleAPI.js 模块中，定义名为 unfollowUserAPI 的接口：

2
3

  1. 在 ArticleDetail.vue 组件中按需导入 unfollowUserAPI 方法：

1
2
  1. 在 ArticleDetail.vue 组件的 methods 节点下声明 setUnfollow 的方法:
methods: {
  // 取关作者
  async setUnfollow() {
    // 1. 调用 API 接口
    const res = await unfollowUserAPI(this.article.aut_id.toString())

    // 2. 判断响应的状态码
    if (res.status === 204) {
      // 2.1 提示用户
      this.$toast.success('取关成功！')
      // 2.2 手动更改关注的状态
      this.article.is_followed = false
    }
  }
}

# 8.8 分支的合并与提交

  1. 将修改过后的文件加入暂存区，并进行本地的 commit 提交：
git add .
git commit -m "完成了文章详情的功能"
1
2
  1. 将本地的 art-detail 分支首次推送到 Gitee 仓库中：
git push -u origin art-detail
1
  1. 将本地的 art-detail 分支合并到本地的 master 主分支，并推送 master 分支到 Gitee 仓库：
git checkout master
git merge art-detail
git push
1
2
3
  1. 删除本地的 art-detail 子分支：
git branch -d art-detail
1
  1. 基于 master 主分支，新建 art-cmt 分支，准备开发文章评论相关的功能：
git checkout -b art-cmt
